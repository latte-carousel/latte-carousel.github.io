Get Started

First include both stylesheet and script files:

<link rel="stylesheet" href="latte-carousel.min.css">
<script src="latte-carousel.min.js"></script>

Then add the following HTML for each carousel:

<div class="latte-carousel">
    <div class="latte-content">
        <div class="latte-stage">
            <div class="latte-item">...</div>
            <div class="latte-item">...</div>
            <div class="latte-item">...</div>
            <div class="latte-item">...</div>
        </div>
    </div>
</div>

Finally create a new instance passing the carousel element and custom options:

var root = document.querySelector('.latte-carousel');

var options = {
    count: 3,
    touch: true,
    buttons: true,
    dots: true,
    rewind: true,
    autoplay: 0,
    responsive: {
        "0":    { count: 1.5, buttons: false },
        "480":  { count: 2.5, buttons: false },
        "768":  { count: 3,   touch: false },
        "1440": { count: 4,   touch: false }
    }
};

var carousel = new latte.Carousel(root, options);

You can also trigger events to change position:

carousel.trigger('next');
carousel.trigger('previous');
